مشروع مراقبة مستوى المياه

حل بسيط وفعال لمراقبة مستويات المياه في الخزانات للمنازل في كردستان

لماذا هذا المشروع مهم؟

يحل مشاكل المياه

في كردستان، نواجه مشاكل متكررة مع المياه. في كثير من الأحيان تفرغ الخزانات دون أن ندرك ذلك، أو في معظم المنازل يتم إهدار الكثير من المياه.

تنبيهات صوتية

عندما يكون مستوى المياه أقل من 15٪ أو أكثر من 80٪، يصدر المشروع صوت تنبيه.

عرض مستوى المياه

تعرض الشاشة نسبة المياه وخط يوضح مستوى الامتلاء، مما يتيح لك معرفة مستوى المياه مباشرةً.

تركيب سهل

بخطوات بسيطة، يمكنك تركيب المشروع وتشغيله فوراً.

الفوائد في كردستان

  • للمنازل: لم يعد من الضروري الصعود إلى السطح للتحقق من المياه
  • للزراعة: اعرف متى ينفد الماء للري
  • في فصل الصيف: يساعدك على تجنب إهدار الماء

كيف يعمل المشروع؟

  1. يقيس المستشعر فوق الصوتي المسافة إلى سطح الماء
  2. إذا كان مستوى المياه أقل من 15٪ أو أكثر من 80٪، يصدر صوت إنذار
  3. تعرض الشاشة نسبة المياه وخط يوضح مستوى الامتلاء
  4. يتم تحديث مستوى المياه على الشاشة بشكل مستمر

مخطط المشروع

المكونات وطريقة التوصيل

المكونات:

  • أردوينو (أونو أو نانو)
  • شاشة LCD 16x2 (نوع I2C)
  • المستشعر فوق الصوتي HC-SR04
  • جهاز التنبيه (بازر)
  • أسلاك التوصيل (ذكر إلى أنثى)
  • لوحة التوصيل التجريبية

طريقة التوصيل:

المكون الطرف التوصيل إلى الأردوينو
شاشة LCD (I2C) VCC 5V
GND GND
SDA A4
SCL A5
المستشعر فوق الصوتي HC-SR04 VCC 5V
GND GND
TRIG الطرف 9
ECHO الطرف 10
جهاز التنبيه (بازر) الطرف (+) الطرف 8
الطرف (-) GND

فيديو كيفية عمل المشروع

كود الأردوينو 
#include <Wire.h>
#include <LiquidCrystal_I2C.h>
                            
#define TRIG_PIN 9
#define ECHO_PIN 10
#define BUZZER_PIN 8
                            
LiquidCrystal_I2C lcd(0x27, 16, 2);
                 
const float TANK_DEPTH = 80.0; 
const float MIN_ALERT_LEVEL = 15.0; 
const float MAX_ALERT_LEVEL = 80.0; 
                            
float waterLevel = 0.0;
float waterPercentage = 0.0;
unsigned long lastBuzzerPatternTime = 0;
int alarmPattern = 0;  
                            
void setup() {
    Serial.begin(9600);
                   
    lcd.init();
    lcd.backlight();
                              
    pinMode(TRIG_PIN, OUTPUT);
    pinMode(ECHO_PIN, INPUT);
    pinMode(BUZZER_PIN, OUTPUT);
                              
    lcd.setCursor(0, 0);
    lcd.print("Water Level");
    lcd.setCursor(0, 1);
    lcd.print("Monitoring System");
    delay(2000);
    lcd.clear();
}
                            
void loop() {
    float distance = measureDistance();
                              
    waterLevel = TANK_DEPTH - distance;
                              
    if (waterLevel < 0) {
        waterLevel = 0;
    }
                              
    waterPercentage = (waterLevel / TANK_DEPTH) * 100.0;
                              
    updateLCD();
                              
    manageLoudAlarm();
                              
    Serial.print("Distance: ");
    Serial.print(distance);
    Serial.print(" cm, Water Level: ");
    Serial.print(waterLevel);
    Serial.print(" cm (");
    Serial.print(waterPercentage);
    Serial.println("%)");
                              
    delay(50);
}
                            
float measureDistance() {
    digitalWrite(TRIG_PIN, LOW);
    delayMicroseconds(2);
                              
    digitalWrite(TRIG_PIN, HIGH);
    delayMicroseconds(10);
    digitalWrite(TRIG_PIN, LOW);
                              
    long duration = pulseIn(ECHO_PIN, HIGH);
                              
    float distance = duration * 0.034 / 2;
                              
    if (distance > TANK_DEPTH) {
        distance = TANK_DEPTH;
    }
                              
    return distance;
}
                            
void updateLCD() {
    static unsigned long lastLCDUpdate = 0;
    if (millis() - lastLCDUpdate < 500) {
        return;
    }
    lastLCDUpdate = millis();
                              
    lcd.clear();
                              
    lcd.setCursor(0, 0);
    lcd.print("Water Level: ");
    lcd.print((int)waterPercentage);
    lcd.print("%");
                              
    lcd.setCursor(0, 1);
    lcd.print("[");
                              
    int bars = map((int)waterPercentage, 0, 100, 0, 14);
                              
    for (int i = 0; i < 14; i++) {
        if (i < bars) {
            lcd.print("=");
        } 
        else {
            lcd.print(" ");
        }
    }
                              
    lcd.print("]");
                              
    if (waterPercentage <= MIN_ALERT_LEVEL || waterPercentage >= MAX_ALERT_LEVEL) {
        lcd.setCursor(15, 0);
        lcd.print((millis() / 250) % 2 ? "!" : "*"); 
    }
}
                            
void manageLoudAlarm() {
    bool isAlertCondition = (waterPercentage <= MIN_ALERT_LEVEL || waterPercentage >= MAX_ALERT_LEVEL);
                              
    if (isAlertCondition) {
        unsigned long currentTime = millis();
                                
        if (currentTime - lastBuzzerPatternTime >= 100) {
            lastBuzzerPatternTime = currentTime;
            alarmPattern++;
                                  
            switch (alarmPattern % 16) {
                case 0:
                case 1:
                case 2:
                    tone(BUZZER_PIN, 2000, 50);
                    break;
                case 3:
                case 4:
                    noTone(BUZZER_PIN);
                    break;
                case 5:
                case 6:
                case 7:
                    tone(BUZZER_PIN, 1500, 50);
                    break;
                case 8:
                    noTone(BUZZER_PIN);
                    break;
                case 9:
                case 10:
                case 11:
                    tone(BUZZER_PIN, 1000, 50);
                    break;
                case 12:
                    noTone(BUZZER_PIN);
                    break;
                case 13:
                case 14:
                    tone(BUZZER_PIN, 2500, 50);
                    break;
                case 15:
                    for (int i = 1000; i < 3000; i += 100) {
                        tone(BUZZER_PIN, i, 15);
                        delay(10);
                    }
                    break;
            }
        }
    } 
    else {
        noTone(BUZZER_PIN);
    }
}
